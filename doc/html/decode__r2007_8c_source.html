56 char *
57 read_system_page(Bit_Chain *dat, int64_t size_comp, int64_t size_uncomp,
58  int64_t repeat_count)
59 {
60  int i;
61  int64_t pesize; // Pre RS encoded size
62  int64_t block_count; // Number of RS encoded blocks
63  int64_t page_size;
64  char *rsdata; // RS encoded data
65  char *pedata; // Pre RS encoded data
66  char *data; // The data RS unencoded and uncompressed
67 
68  // Round to a multiple of 8
69  pesize = ((size_comp + 7) & ~7) * repeat_count;
70 
71  // Divide pre encoded size by RS k-value (239)
72  block_count = (pesize + 238) / 239;
73 
74  // Multiply with codeword size (255) and round to a multiple of 8
75  page_size = (block_count * 255 + 7) & ~7;
76 
77  data = (char*) malloc(size_uncomp + page_size);
78 
79  if (data == 0)
80  {
81  //TODO: report error
82  return 0;
83  }
84  rsdata = &data[size_uncomp];
85 
86  for (i = 0; i < page_size; i++)
87  rsdata[i] = bit_read_RC(dat);
88 
89  pedata = decode_rs(rsdata, block_count, 239);
90 
91  if (size_comp < size_uncomp)
92  decompress_R2007(data, size_uncomp, pedata, size_comp);
93  else
94  memcpy(data, pedata, size_uncomp);
95 
96  free(pedata);
97  return data;
98 }

355 string_stream_init(Bit_Chain *sstream, Bit_Chain *dat,
356  unsigned long int bitpos, int check_present_bit)
357 {
358  int strsize, hisize;
359 
360  uint64_t mem_byte = dat->byte;
361  uint8_t mem_bit = dat->bit;
362 
363  if (check_present_bit)
364  {
365  dat->byte = ((bitpos-8) >> 3);
366  dat->bit = ((bitpos-8) & 7);
367 
368  if (bit_read_RC(dat) == 1)
369  {
370  bitpos -= 17;
371  dat->byte = (bitpos >> 3);
372  dat->bit = (bitpos & 7);
373  }
374  else
375  {
376  // no string stream present
377  dat->byte = mem_byte;
378  dat->bit = mem_bit;
379  return dat;
380  }
381  }
382  else
383  {
384  bitpos += 0x8F; // 8 * 16 + 15
385  dat->byte = (bitpos >> 3);
386  dat->bit = (bitpos & 7);
387  }
388  strsize = bit_read_RS(dat);
389 
390  if (strsize & 0x8000)
391  {
392  strsize &= 0x7FFF; //~0x8000;
393  bitpos -= 16;
394 
395  dat->byte = (bitpos >> 3);
396  dat->bit = (bitpos & 7);
397 
398  hisize = bit_read_RS(dat);
399  strsize |= (hisize << 15);
400  }
401  bitpos -= strsize;
402 
403  sstream->byte = bitpos >> 3;
404  sstream->bit = bitpos & 7;
405  sstream->chain = dat->chain;
406  sstream->size = dat->size;
407  sstream->version = dat->version;
408 
409  dat->byte = mem_byte;
410  dat->bit = mem_bit;
411  return sstream;
